import streamlit as st import pandas as pd import numpy as np import plotly.express as px # Configuração da página do Streamlit st.set_page_config( page_title="Agro Forte, Futuro Sustentável", page_icon="🌱", layout="wide" ) # Título Principal e Contexto st.title("🌱 Agro Forte, Futuro Sustentável") st.subheader("O Equilíbrio entre a Produção de Alimentos e a Conservação Ambiental") st.markdown(""" Este painel interativo demonstra como métricas de tecnologia e manejo sustentável podem otimizar a produção agrícola mantendo a pegada ecológica sob controle. """) st.divider() # --- SIDEBAR: Filtros de Simulação --- st.sidebar.header("Configurações da Fazenda") tamanho_fazenda = st.sidebar.slider("Área Cultivada (Hectares)", 100, 5000, 1200) tecnologia = st.sidebar.selectbox( "Nível de Tecnologia de Manejo", ["Tradicional", "Integrada (ILPF)", "Agricultura de Precisão / Regenerativa"] ) # --- LÓGICA DE DADOS (Simulação baseada nos filtros) --- np.random.seed(42) meses = ["Jan", "Fev", "Mar", "Abr", "Mai", "Jun", "Jul", "Ago", "Set", "Out", "Nov", "Dez"] # Ajuste de multiplicadores baseado na tecnologia escolhida if tecnologia == "Tradicional": mult_prod = 1.0 mult_agua = 1.3 mult_carbono = -0.2 elif tecnologia == "Integrada (ILPF)": mult_prod = 1.2 mult_agua = 0.9 mult_carbono = 0.5 else: # Regenerativa mult_prod = 1.35 mult_agua = 0.7 mult_carbono = 1.2 # Gerando dados fictícios plausíveis dados = { "Mês": meses, "Produção (Toneladas)": np.round(np.random.normal(50, 5, 12) * (tamanho_fazenda / 100) * mult_prod, 1), "Consumo de Água (Liters/kg produzido)": np.round(np.random.normal(250, 15, 12) * mult_agua, 1), "Créditos de Carbono Gerados (Saldos)": np.round(np.random.normal(10, 2, 12) * (tamanho_fazenda / 100) * mult_carbono, 1) } df = pd.DataFrame(dados) # --- METRICAS PRINCIPAIS --- col1, col2, col3 = st.columns(3) with col1: total_prod = df["Produção (Toneladas)"].sum() st.metric(label="Produção Total Anual", value=f"{total_prod:,.1f} Ton") with col2: media_agua = df["Consumo de Água (Liters/kg produzido)"].mean() st.metric(label="Eficiência Hídrica Média", value=f"{media_agua:.1f} L/kg", delta=f"{-10 if tecnologia != 'Tradicional' else 5}%", delta_color="inverse") with col3: total_carbono = df["Créditos de Carbono Gerados (Saldos)"].sum() st.metric(label="Balanço de Carbono (CO2e)", value=f"{total_carbono:+,.1f} t", delta="Sequestro de Carbono" if total_carbono > 0 else "Emissão") st.divider() # --- GRÁFICOS INTERATIVOS --- col_graf1, col_graf2 = st.columns(2) with col_graf1: st.write("### 📈 Evolução da Produção vs. Sustentabilidade") fig_prod = px.line(df, x="Mês", y="Produção (Toneladas)", title="Produção Agrícola Mensal", markers=True, color_discrete_sequence=["#2E7D32"]) st.plotly_chart(fig_prod, use_container_width=True) with col_graf2: st.write("### 🍃 Balanço de Carbono Mensal") # Define cor dinâmica se for positivo (verde) ou negativo (vermelho) df['Status Carbono'] = df['Créditos de Carbono Gerados (Saldos)'].apply(lambda x: 'Sequestro' if x >= 0 else 'Emissão') fig_carb = px.bar(df, x="Mês", y="Créditos de Carbono Gerados (Saldos)", color='Status Carbono', title="Créditos de Carbono Gerados / Retidos", color_discrete_map={'Sequestro': '#4CAF50', 'Emissão': '#E53935'}) st.plotly_chart(fig_carb, use_container_width=True) # --- SEÇÃO INFORMATIVA --- st.subheader("💡 Pilares do Agro Sustentável Demonstrados") st.markdown(""" * **Eficiência Hídrica:** Tecnologias como gotejamento e sensores de umidade de solo reduzem drasticamente o desperdício de água. * **ILPF (Integração Lavoura-Pecuária-Floresta):** Aumenta a biodiversidade do solo e ajuda a neutralizar as emissões de gases estufa da propriedade. * **Decisão Baseada em Dados:** Monitorar índices de produtividade em paralelo com índices ecológicos prova que a sustentabilidade impulsiona a lucratividade do produtor rural. """)
