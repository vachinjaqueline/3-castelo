import streamlit as st import pandas as pd import numpy as np import plotly.express as px # Configuração da página do Streamlit st.set_page_config( page_title="Agro Forte, Futuro Sustentável", page_icon="🌱", layout="wide" ) # Título Principal e Contexto st.title("🌱 Agro Forte, Futuro Sustentável") st.subheader("O Equilíbrio entre a Produção de Alimentos e a Conservação Ambiental") st.markdown(""" Este painel interativo demonstra como métricas de tecnologia e manejo sustentável podem otimizar a produção agrícola mantendo a pegada ecológica sob controle. """) st.divider() # --- SIDEBAR: Filtros de Simulação --- st.sidebar.header("Configurações da Fazenda") tamanho_fazenda = st.sidebar.slider("Área Cultivada (Hectares)", 100, 5000, 1200) tecnologia = st.sidebar.selectbox( "Nível de Tecnologia de Manejo", ["Tradicional", "Integrada (ILPF)", "Agricultura de Precisão / Regenerativa"] ) # --- LÓGICA DE DADOS (Simulação baseada nos filtros) --- np.random.seed(42) meses = ["Jan", "Fev", "Mar", "Abr", "Mai", "Jun", "Jul", "Ago", "Set", "Out", "Nov", "Dez"] # Ajuste de multiplicadores baseado na tecnologia escolhida if tecnologia == "Tradicional": mult_prod = 1.0 mult_agua = 1.3 mult_carbono = -0.2 elif tecnologia == "Integrada (ILPF)": mult_prod = 1.2 mult_agua = 0.9 mult_carbono = 0.5 else: # Regenerativa mult_prod = 1.35 mult_agua = 0.7 mult_carbono = 1.2 # Gerando dados fictícios plausíveis dados = { "Mês": meses, "Produção (Toneladas)": np.round(np.random.normal(50, 5, 12) * (tamanho_fazenda / 100) * mult_prod, 1), "Consumo de Água (Liters/kg produzido)": np.round(np.random.normal(250, 15, 12) * mult_agua, 1), "Créditos de Carbono Gerados (Saldos)": np.round(np.random.normal(10, 2, 12) * (tamanho_fazenda / 100) * mult_carbono, 1) } df = pd.DataFrame(dados) # --- METRICAS PRINCIPAIS --- col1, col2, col3 = st.columns(3) with col1: total_prod = df["Produção (Toneladas)"].sum() st.metric(label="Produção Total Anual", value=f"{total_prod:,.1f} Ton")
